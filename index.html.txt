<script>
const ENDPOINT = 'https://script.google.com/macros/s/AKfycbzOX666wRMh6iDxgIAgoI5I8BUJv4NzsIXDf0QaPJh-3wF-JxFCEHronVU7C9mqGW8/exec';

function validarCPF(cpf) {
  return /^\d{3}\.\d{3}\.\d{3}-\d{2}$/.test(cpf);
}

async function inscrever() {
  const nome = nomeEl.value.trim();
  const cpf = cpfEl.value.trim();
  const endereco = enderecoEl.value.trim();

  if (!nome || !cpf || !endereco) {
    alert('Preencha todos os campos');
    return;
  }

  if (!validarCPF(cpf)) {
    alert('CPF inválido. Use o formato 000.000.000-00');
    return;
  }

  const dados = { nome, cpf, endereco };

  try {
    await fetch(ENDPOINT, {
      method: 'POST',
      body: JSON.stringify(dados)
    });

    msg.innerText = 'Inscrição realizada!';
    nomeEl.value = cpfEl.value = enderecoEl.value = '';

  } catch (e) {
    alert('Erro ao enviar inscrição. Tente novamente.');
  }
}

const nomeEl = document.getElementById('nome');
const cpfEl = document.getElementById('cpf');
const enderecoEl = document.getElementById('endereco');
const msg = document.getElementById('msg');
</script>
